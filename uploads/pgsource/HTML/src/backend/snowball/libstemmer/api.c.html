extern void SN_close_env(struct SN_env * z, int S_size) 
{ 
    if (z == NULL) return; 
    if (S_size) 
    { 
        int i; 
        for (i = 0; i < S_size; i++) 
        { 
            lose_s(z->S[i]); 
        } 
        free(z->S); 
    } 
    free(z->I); 
    free(z->B); 
    if (z->p) lose_s(z->p); 
    free(z); 
}
