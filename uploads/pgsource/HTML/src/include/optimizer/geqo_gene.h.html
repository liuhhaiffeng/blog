#define GEQO_GENE_H

   is o.k. for Gene; so don't change it! */

typedef struct Chromosome 
{ 
    Gene       *string; 
    Cost        worth; 
} Chromosome;

{ 
    Chromosome *data; 
    int         size; 
    int         string_length; 
} Pool;
