/* 
 * Some compatibility functions

#define EWOULDBLOCK WSAEWOULDBLOCK 
#endif 
#ifndef ECONNRESET 
#define ECONNRESET WSAECONNRESET 
#endif 
#ifndef EINPROGRESS

#endif
