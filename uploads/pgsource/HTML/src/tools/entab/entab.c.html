#define TRUE    1

extern char *optarg;

static void 
output_accumulated_spaces(int *prv_spaces, char **dst) 
{ 
    for (; *prv_spaces > 0; (*prv_spaces)--) 
        *((*dst)++) = ' '; 
} 
 
 
static void 
trim_trailing_whitespace(int *prv_spaces, char **dst, char *out_line) 
{ 
    while (*dst > out_line && 
           (*((*dst) - 1) == ' ' || *((*dst) - 1) == '\t')) 
        (*dst)--; 
    *prv_spaces = 0; 
}
